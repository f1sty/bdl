# BDL (bipolar disorder language) draft

## Literals:

- identifier ('a'..'z' 'A'..'Z' '_')
- string ("example")
- i64 (123)
- f64 (17.22)
- # (define function)
- @ (lambda function)
- ! (define constant)
- ? (defive variable)
- : (return)
- -> (include/arrow)
- $ (reference)
- % (dereference)

## Keywords:

- if
- else
- while
- or
- xor
- and
- not
- for
- nil
- true
- false
- print

## Other tokens:

- [ ] (list/array?)
- ( ) (brackets)
- { } (map/hash)
- = (assign)
- == >= <= > < (comparison)
- += -= *= /= (compact arithmetics)
- + - * / (arithmetics)
- | (block marker)
- EOF (end of file)
